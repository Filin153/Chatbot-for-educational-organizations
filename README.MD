# Подстановка секретных переменных в окружение (.env)
    BOT_TOKEN - токен вашего бота
    POSTGRES_DB - название базы данных
    POSTGRES_USER - имя пользователя
    POSTGRES_PASSWORD - пароль
    POSTGRES_HOST - хост

# Запуск docker-compose
    docker-compose up -d
